\section{Sample algorithm}
\begin{algorithm}
	\caption{Kruskal's Algorithm}\label{alg:kruskal}
	\begin{algorithmic}[1]
		\Procedure{MakeSet}{$v$}
			\State Create new set containing $v$
		\EndProcedure \\
		
		\Function{FindSet}{$v$} 
			\State\Return a set containing $v$
		\EndFunction \\
		
		\Procedure{Union}{$u$,$v$}
			\State Unites the set that contain $u$ and $v$ into a new set
		\EndProcedure \\
		
		\Function{Kruskal}{$V,E,w$}
			\State $A \gets \{\}$
			\For {each vertex $v$ in $V$}
				\State MakeSet($v$)
			\EndFor
			\State Arrange $E$ in increasing costs, ordered by $w$
			\For {each ($u$,$v$) taken from the sorted list}
				\If {FindSet($u$) $\neq$ FindSet($v$)}
					\State $A \gets A \cup \{(u,v)\}$
					\State Union($u,v$)
				\EndIf
			\EndFor
			\State \Return A
		\EndFunction
	\end{algorithmic}
\end{algorithm}

import numpy as np

def incmatrix(genl1,genl2):
	m = len(genl1)
	n = len(genl2)
	M = None #to become the incidence matrix
	VT = np.zeros((n*m,1), int)  #dummy variable

	#compute the bitwise xor matrix
	M1 = bitxormatrix(genl1)
	M2 = np.triu(bitxormatrix(genl2),1) 

	for i in range(m-1):
		for j in range(i+1, m):
			[r,c] = np.where(M2 == M1[i,j])
			for k in range(len(r)):
				VT[(i)*n + r[k]] = 1;
				VT[(i)*n + c[k]] = 1;
				VT[(j)*n + r[k]] = 1;
				VT[(j)*n + c[k]] = 1;

	if M is None:
		M = np.copy(VT)
	else:
		M = np.concatenate((M, VT), 1)

	VT = np.zeros((n*m,1), int)

	return M
\end{lstlisting}
